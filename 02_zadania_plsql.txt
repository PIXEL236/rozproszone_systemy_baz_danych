END;

DECLARE
  v_umowy NUMBER;
BEGIN
  select 
    SUM(r.CENA)
  into v_umowy
  from UMOWY u
  inner join KURSYSIEDZIBA k on k.KURS_ID = u.KURS_ID
  inner join RODZAJESIEDZIBA r on r.RODZAJ_ID = k.RODZAJ_ID
  where u.MIASTO = 'BYDGOSZCZ';
  
  DBMS_OUTPUT.PUT_LINE('Liczba umów: ' || v_umowy);
END;

DECLARE
  v_umowy NUMBER;
  v_miasto  VARCHAR2(100);
BEGIN
  v_miasto := 'BYDGOSZCZ';
  
  select 
    SUM(r.CENA)
  into v_umowy
  from UMOWY u
  inner join KURSYSIEDZIBA k on k.KURS_ID = u.KURS_ID
  inner join RODZAJESIEDZIBA r on r.RODZAJ_ID = k.RODZAJ_ID
  where u.MIASTO = v_miasto;
  
  IF v_umowy = 0 THEN
    DBMS_OUTPUT.PUT_LINE('Brak umów');
  ELSIF v_umowy < 50 THEN
    DBMS_OUTPUT.PUT_LINE('Mała liczba umów');
  ELSIF v_umowy > 50 and v_umowy < 100 THEN
    DBMS_OUTPUT.PUT_LINE('Średnia liczba umów');
  ELSE
    DBMS_OUTPUT.PUT_LINE('Duża liczba umów');
  END IF;
END;

--Zadanie 5
CREATE OR REPLACE PROCEDURE raport_umow_miasto(p_miasto IN VARCHAR2)
AS
BEGIN
  DECLARE
    v_umowy NUMBER;
    v_wartoscumow NUMBER;
    v_sredniawartosc NUMBER;
  BEGIN
    select 
      COUNT(*)
    into v_umowy
    from UMOWY u
    inner join KURSY k on k.KURS_ID = u.KURS_ID
    inner join RODZAJE r on r.RODZAJ_ID = k.RODZAJ_ID
    where u.MIASTO = p_miasto;
    
    select 
      SUM(r.CENA)
    into v_wartoscumow
    from UMOWY u
    inner join KURSY k on k.KURS_ID = u.KURS_ID
    inner join RODZAJE r on r.RODZAJ_ID = k.RODZAJ_ID
    where u.MIASTO = p_miasto;
 
    select 
      AVG(r.CENA)
    into v_sredniawartosc
    from UMOWY u
    inner join KURSY k on k.KURS_ID = u.KURS_ID
    inner join RODZAJE r on r.RODZAJ_ID = k.RODZAJ_ID
    where u.MIASTO = p_miasto;
    
    DBMS_OUTPUT.PUT_LINE('Miasto: ' || p_miasto);
    DBMS_OUTPUT.PUT_LINE('Ilość umów: ' || v_umowy);
    DBMS_OUTPUT.PUT_LINE('Wartość umów: ' || v_wartoscumow);
    DBMS_OUTPUT.PUT_LINE('Średnia wartość umowy: ' || v_sredniawartosc);
  END;
END;
/

BEGIN
  raport_umow_miasto('BYDGOSZCZ');
END;
/

--Zadanie 6
CREATE OR REPLACE FUNCTION wartosc_kursu(
  p_kurs_id IN NUMBER
)
RETURN NUMBER
AS
  v_liczba NUMBER;
BEGIN
  SELECT
    r.CENA
  INTO v_liczba
  FROM KURSY k
  inner join RODZAJE r on r.RODZAJ_ID = k.RODZAJ_ID
  WHERE k.KURS_ID = p_kurs_id;

  RETURN v_liczba;
END;
/

--Zadanie 7
CREATE OR REPLACE PROCEDURE pokaz_kursanta(p_kursant_id IN NUMBER)
AS
BEGIN
  DECLARE
    v_imie VARCHAR2;
    v_nazwisko VARCHAR2;
  BEGIN
    SELECT 
      k.IMIE
    INTO v_imie
    FROM KURSANCI k
    WHERE k.KURSANT_ID = p_kursant_id;

    SELECT 
      k.NAZWISKO
    INTO v_nazwisko
    FROM KURSANCI k
    WHERE k.KURSANT_ID = p_kursant_id;

    DBMS_OUTPUT.PUT_LINE(v_imie);
    DBMS_OUTPUT.PUT_LINE(v_nazwisko);

    EXCEPTION
    WHEN NO_DATA_FOUND THEN
      DBMS_OUTPUT.PUT_LINE('Nie znaleziono kursanta.');
  END;
END;
/

--Zadanie 8
DECLARE
  CURSOR c_kursy IS
    select
      u.UMOWA_ID
    , k2.IMIE
    , k2.NAZWISKO
    , r.NAZWA
    , r.CENA
    from UMOWY u
    inner join KURSY k on k.KURS_ID = u.KURS_ID
    inner join RODZAJE r on r.RODZAJ_ID = k.RODZAJ_ID
    inner join KURSANCI k2 on k2.KURSANT_ID =u.KURSANT_ID 
    where u.MIASTO = 'BYDGOSZCZ';

  v_kurs c_kursy%ROWTYPE;
BEGIN
  OPEN c_kursy;

  LOOP
    FETCH c_kursy INTO v_kurs;
    EXIT WHEN c_kursy%NOTFOUND;

    DBMS_OUTPUT.PUT_LINE('Umowa ' || v_kurs || ' | ' v_kurs.IMIE || ' ' || v_kurs.NAZWISKO || ' | ' v_kurs.NAZWA || ' | ' || v_kurs.CENA);
  END LOOP;

  CLOSE c_kursy;
END;
/

--Zadanie 10
CREATE OR REPLACE PROCEDURE raport_uczelni()
AS
BEGIN
  DECLARE
    v_liczba_umow_b NUMBER;
    v_suma_wartosci_b NUMBER;
    v_najdrozszy_kurs_b VARCHAR2;
    v_najpopularniejszy_kurs_b VARCHAR2;

    v_liczba_umow_s NUMBER;
    v_suma_wartosci_s NUMBER;
    v_najdrozszy_kurs_s VARCHAR2;
    v_najpopularniejszy_kurs_s VARCHAR2;

    v_wszystkie_umowy NUMBER;
    v_wartosc_wszystkich_umow NUMBER;
  BEGIN
    SELECT 
      COUNT(*)
    INTO v_liczba_umow_b
    FROM UMOWY u
    WHERE u.MIASTO = 'BYDGOSZCZ';

    select 
      SUM(r.CENA)
    into v_suma_wartosci_b
    from UMOWY u
    inner join KURSY k on k.KURS_ID = u.KURS_ID
    inner join RODZAJ r on r.RODZAJ_ID = k.RODZAJ_ID
    where u.MIASTO = 'BYDGOSZCZ';

    select
      r.NAZWA
    into v_najdrozszy_kurs_b
    from
    (
      select
        k.KURS_ID
      from KURSY k
      inner join RODZAJ r on r.RODZAJ_ID = k.RODZAJ_ID
      order by r.CENA desc
    )k1
    inner join KURSY k2 on k2.KURS_ID = k1.KURS_ID
    inner join RODZAJ r on r.RODZAJ_ID = k2.RODZAJ_ID
    where rownum = 1;

    select
      r.NAZWA
    into v_najpopularniejszy_kurs_b
    from
    (
      select
        k.KURS_ID
      , COUNT(*) Umowy
      from KURSY k
      inner join UMOWY u on u.KURS_ID = k.KURS_ID
      where u.MIASTO = 'BYDGOSZCZ'
      group by k.KURS_ID
      order by COUNT(*) desc
    )k1
    inner join KURSY k2 on k2.KURS_ID = k1.KURS_ID
    inner join RODZAJ r on r.RODZAJ_ID = k2.RODZAJ_ID
    where rownum = 1;


    SELECT 
      COUNT(*)
    INTO v_liczba_umow_s
    FROM UMOWY u
    WHERE u.MIASTO = 'SZCZECIN';

    select 
      SUM(r.CENA)
    into v_suma_wartosci_s
    from UMOWY u
    inner join KURSY@filia k on k.KURS_ID = u.KURS_ID
    inner join RODZAJ@filia r on r.RODZAJ_ID = k.RODZAJ_ID
    where u.MIASTO = 'SZCZECIN';

    select
      r.NAZWA
    into v_najdrozszy_kurs_s
    from
    (
      select
        k.KURS_ID
      from KURSY@filia k
      inner join RODZAJ@filia r on r.RODZAJ_ID = k.RODZAJ_ID
      order by r.CENA desc
    )k1
    inner join KURSY@filia k2 on k2.KURS_ID = k1.KURS_ID
    inner join RODZAJ r on r.RODZAJ_ID = k2.RODZAJ_ID
    where rownum = 1;

    select
      r.NAZWA
    into v_najpopularniejszy_kurs_s
    from
    (
      select
        k.KURS_ID
      , COUNT(*) Umowy
      from KURSY@filia k
      inner join UMOWY u on u.KURS_ID = k.KURS_ID
      where u.MIASTO = 'SZCZECIN'
      group by k.KURS_ID
      order by COUNT(*) desc
    )k1
    inner join KURSY@filia k2 on k2.KURS_ID = k1.KURS_ID
    inner join RODZAJ r on r.RODZAJ_ID = k2.RODZAJ_ID
    where rownum = 1;

    select
      v_liczba_umow_b + v_liczba_umow_s
    into v_wszystkie_umowy;

    select
      v_suma_wartosci_b + v_suma_wartosci_s
    into v_wartosc_wszystkich_umow;

    DBMS_OUTPUT.PUT_LINE('Miasto: BYDGOSZCZ');
    DBMS_OUTPUT.PUT_LINE('Liczba umów:' || v_liczba_umow_b);
    DBMS_OUTPUT.PUT_LINE('Łączna wartość umów: ' || v_suma_wartosci_b);
    DBMS_OUTPUT.PUT_LINE('Najdroższy kurs: ' || v_najdrozszy_kurs_b);
    DBMS_OUTPUT.PUT_LINE('Najdroższy kurs: ' || v_najpopularniejszy_kurs_b);

    DBMS_OUTPUT.PUT_LINE('Miasto: SZCZECIN');
    DBMS_OUTPUT.PUT_LINE('Liczba umów:' || v_liczba_umow_s);
    DBMS_OUTPUT.PUT_LINE('Łączna wartość umów: ' || v_suma_wartosci_s);
    DBMS_OUTPUT.PUT_LINE('Najdroższy kurs: ' || v_najdrozszy_kurs_s);
    DBMS_OUTPUT.PUT_LINE('Najdroższy kurs: ' || v_najpopularniejszy_kurs_s);

    DBMS_OUTPUT.PUT_LINE('PODSUMOWANIE');
    DBMS_OUTPUT.PUT_LINE('Liczba wszystkich  umów:' || v_wszystkie_umowy);
    DBMS_OUTPUT.PUT_LINE('Łączna wartość wszystkich umów: ' || v_wartosc_wszystkich_umow);
  END;
END;
/
