--Zadanie 1
DECLARE
  v_kursanci NUMBER;
  v_kursy NUMBER;
  v_wykladowcy NUMBER;
BEGIN
  select 
    COUNT(*) 
  into v_kursanci
  from KURSANCISIEDZIBA;

  select 
    COUNT(*) 
  into v_kursy
  from KURSANCISIEDZIBA;
  
  select 
    COUNT(*) 
  into v_wykladowcy
  from WYKLADOWCYSIEDZIBA;
  
  DBMS_OUTPUT.PUT_LINE('Liczba kursantów: ' || v_kursanci);
  DBMS_OUTPUT.PUT_LINE('Liczba kursów: ' || v_kursy);
  DBMS_OUTPUT.PUT_LINE('Liczba wykadowców: ' || v_wykladowcy);
END;
/

--Zadanie 2
DECLARE
  v_umowy NUMBER;
BEGIN
  select 
    SUM(r.CENA)
  into v_umowy
  from UMOWYSIEDZIBA u
  inner join KURSYSIEDZIBA k on k.KURS_ID = u.KURS_ID
  inner join RODZAJESIEDZIBA r on r.RODZAJ_ID = k.RODZAJ_ID
  where u.MIASTO = 'BYDGOSZCZ';
  
  DBMS_OUTPUT.PUT_LINE('Liczba umów: ' || v_umowy);
END;
/

--Zadanie 3
DECLARE
  v_umowy NUMBER;
  v_miasto  VARCHAR2(100);
BEGIN
  v_miasto := 'BYDGOSZCZ';
  
  select 
    SUM(r.CENA)
  into v_umowy
  from UMOWYSIEDZIBA u
  inner join KURSYSIEDZIBA k on k.KURS_ID = u.KURS_ID
  inner join RODZAJESIEDZIBA r on r.RODZAJ_ID = k.RODZAJ_ID
  where u.MIASTO = v_miasto;
  
  IF v_umowy = 0 THEN
    DBMS_OUTPUT.PUT_LINE('Brak umów');
  ELSIF v_umowy < 50 THEN
    DBMS_OUTPUT.PUT_LINE('Mała liczba umów');
  ELSIF v_umowy > 50 and v_umowy < 100 THEN
    DBMS_OUTPUT.PUT_LINE('Średnia liczba umów');
  ELSE
    DBMS_OUTPUT.PUT_LINE('Duża liczba umów');
  END IF;
END;
/

  END LOOP;
END;

--Zadanie 5
CREATE OR REPLACE PROCEDURE raport_umow_miasto(p_miasto IN VARCHAR2)
AS
BEGIN
  DECLARE
    v_umowy NUMBER;
    v_wartoscumow NUMBER;
    v_sredniawartosc NUMBER;
  BEGIN
    select 
      COUNT(*)
    into v_umowy
    from UMOWY u
    inner join KURSY k on k.KURS_ID = u.KURS_ID
    inner join RODZAJE r on r.RODZAJ_ID = k.RODZAJ_ID
    where u.MIASTO = p_miasto;
    
    select 
      SUM(r.CENA)
    into v_wartoscumow
    from UMOWY u
    inner join KURSY k on k.KURS_ID = u.KURS_ID
    inner join RODZAJE r on r.RODZAJ_ID = k.RODZAJ_ID
    where u.MIASTO = p_miasto;
 
    select 
      AVG(r.CENA)
    into v_sredniawartosc
    from UMOWY u
    inner join KURSY k on k.KURS_ID = u.KURS_ID
    inner join RODZAJE r on r.RODZAJ_ID = k.RODZAJ_ID
    where u.MIASTO = p_miasto;
    
    DBMS_OUTPUT.PUT_LINE('Miasto: ' || p_miasto);
    DBMS_OUTPUT.PUT_LINE('Ilość umów: ' || v_umowy);
    DBMS_OUTPUT.PUT_LINE('Wartość umów: ' || v_wartoscumow);
    DBMS_OUTPUT.PUT_LINE('Średnia wartość umowy: ' || v_sredniawartosc);
  END;
END;
/

BEGIN
  raport_umow_miasto('BYDGOSZCZ');
END;
/
