--Zadanie 6
CREATE MATERIALIZED VIEW REP_godz_wykladowcy_godziny
REFRESH COMPLETE
START WITH Last_Day(SYSDATE)
NEXT SYSDATE + 1/24
AS
select
  w.IMIE
, w.NAZWISKO
, SUM(r.GODZ) GODZINY
from WYKLADOWCY@filia w
inner join KURSY@filia k on k.WYKLADOWCA_ID = w.WYKLADOWCA_ID
inner join RODZAJE@filia r on r.RODZAJ_ID = k.RODZAJ_ID
group by w.IMIE, w.NAZWISKO
;

--Zadanie 7
CREATE MATERIALIZED VIEW REP_kursy
BUILD IMMEDIATE
REFRESH COMPLETE
START WITH SYSDATE
NEXT SYSDATE + 7
AS
select
  r.NAZWA
, w.IMIE
, w.NAZWISKO
, r.GODZ
, r.CENA
from KURSY@filia k
inner join WYKLADOWCY@filia w on w.WYKLADOWCA_ID = k.WYKLADOWCA_ID
inner join RODZAJE@filia r on r.RODZAJ_ID = k.RODZAJ_ID
;

--Zadanie 8
CREATE VIEW wykladowcyAll AS
select
  r.NAZWA
, w.IMIE
, w.NAZWISKO
, r.GODZ
, r.CENA
from KURSY@filia k
inner join WYKLADOWCY@filia w on w.WYKLADOWCA_ID = k.WYKLADOWCA_ID
inner join RODZAJE@filia r on r.RODZAJ_ID = k.RODZAJ_ID
union all
select
  k.NAZWA
, k.IMIE
, k.NAZWISKO
, k.GODZ
, k.CENA
from REP_kursy k
;
