--Zadanie 6
CREATE VIEW kursanciAll AS
select
  ks.IMIE
, ks.NAZWISKO
from kursanciSiedziba ks
union all
select
  kf.IMIE
, kf.NAZWISKO
from kursanciFilia kf;

CREATE VIEW wykladowcyAll AS
select
  ks.IMIE
, ks.NAZWISKO
from wykladowcySiedziba ks
union all
select
  kf.IMIE
, kf.NAZWISKO
from wykladowcyFilia kf;

--Zadanie 7
CREATE VIEW kursyAll AS
select
  ks.KURS_ID
, ws.IMIE
, ws.NAZWISKO
, rs.NAZWA
, COUNT(*) ILOSC_KURSANTOW
from kursySiedziba ks
inner join wykladowcySiedziba ws on ws.WYKLADOWCA_ID = ks.WYKLADOWCA_ID
inner join umowySiedziba us on us.KURS_ID = ks.KURS_ID
inner join rodzajeSiedziba rs on rs.RODZAJ_ID = ks.RODZAJ_ID
inner join kursanciSiedziba ks2 on ks2.KURSANT_ID = us.KURSANT_ID
group by ks.KURS_ID, ws.IMIE, ws.NAZWISKO, rs.NAZWA
union all
select
  ks.KURS_ID
, ws.IMIE
, ws.NAZWISKO
, rs.NAZWA
, COUNT(*) ILOSC_KURSANTOW
from kursyFilia ks
inner join wykladowcyFilia ws on ws.WYKLADOWCA_ID = ks.WYKLADOWCA_ID
inner join umowyFilia us on us.KURS_ID = ks.KURS_ID
inner join rodzajeFilia rs on rs.RODZAJ_ID = ks.RODZAJ_ID
inner join kursanciFilia ks2 on ks2.KURSANT_ID = us.KURSANT_ID
group by ks.KURS_ID, ws.IMIE, ws.NAZWISKO, rs.NAZWA
;

inner join umowyFilia us on us.KURS_ID = ks.KURS_ID
inner join rodzajeFilia rs on rs.RODZAJ_ID = ks.RODZAJ_ID
)a
;
